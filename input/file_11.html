基础测试案例和异步代码测试案例

展示如何使用Vitest进行基础函数测试和异步代码测试

functions

基础函数测试

使用Vitest测试简单函数，验证输入输出是否符合预期

// math.js
export const sum = (a, b) => a + b;
export const multiply = (a, b) => a * b;
export const isEven = (num) => num % 2 === 0;

// math.test.js
import { describe, test, expect } from 'vitest';
import { sum, multiply, isEven } from './math.js';

describe('Math functions', () => {
  test('sum adds two numbers', () => {
    expect(sum(1, 2)).toBe(3);
    expect(sum(-1, 1)).toBe(0);
  });

  test('multiply multiplies two numbers', () => {
    expect(multiply(3, 4)).toBe(12);
  });

  test('isEven checks if number is even', () => {
    expect(isEven(4)).toBe(true);
    expect(isEven(5)).toBe(false);
  });
});

异步代码测试

测试异步操作，如Promise、async/await和定时器

// api.js
export const fetchData = () => {
  return new Promise((resolve) => {
    setTimeout(() => {
      resolve({ data: 'test data' });
    }, 1000);
  });
};

// api.test.js
import { test, expect, vi } from 'vitest';
import { fetchData } from './api.js';

// 测试Promise
test('fetchData returns data', async () => {
  const result = await fetchData();
  expect(result).toEqual({ data: 'test data' });
});

// 使用vi.useFakeTimers模拟定时器
test('fetchData with fake timers', () => {
  vi.useFakeTimers();
  const promise = fetchData();

  // 快进时间
  vi.advanceTimersByTime(1000);

  return promise.then(result => {
    expect(result).toEqual({ data: 'test data' });
  });
});
