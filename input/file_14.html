photo_camera

组件快照测试

捕获组件渲染输出的快照，用于跟踪UI变化，确保组件渲染结果符合预期

// Button.jsx
import React from 'react';

function Button({ children, variant = 'primary' }) {
  return (
    <button className={`btn btn-${variant}`}>
      {children}
    </button>
  );
}

// Button.test.jsx
import { render } from '@testing-library/react';
import { test, expect } from 'vitest';
import Button from './Button.jsx';

test('renders primary button correctly', () => {
  const { container } = render(<Button>Click me</Button>);
  expect(container).toMatchSnapshot();
});

test('renders secondary button correctly', () => {
  const { container } = render(
    <Button variant={"secondary"}>Save</Button>
  );
  expect(container).toMatchSnapshot();
});

数据快照测试

捕获数据结构的快照，验证函数输出或API响应是否符合预期格式

// userUtils.js
export const formatUser = (user) => {
  return {
    id: user.id,
    name: `${user.firstName} ${user.lastName}`,
    email: user.email.toLowerCase(),
    createdAt: new Date().toISOString()
  };
};

// userUtils.test.js
import { test, expect, vi } from 'vitest';
import { formatUser } from './userUtils.js';

test('formats user data correctly', () => {
  // 模拟日期
  vi.setSystemTime(new Date('2023-01-01'));

  const user = {
    id: 1,
    firstName: 'John',
    lastName: 'Doe',
    email: 'JOHN@EXAMPLE.COM'
  };

  const formattedUser = formatUser(user);
  // 使用内联快照
  expect(formattedUser).toMatchInlineSnapshot({
    id: 1,
    name: 'John Doe',
    email: 'john@example.com',
    createdAt: '2023-01-01T00:00:00.000Z'
  });
});
